//#btnを取得
let btn = document.getElementById("btn");

btn.addEventListener('click', function(){
  //1から5までの乱数を生成
  let rand = Math.floor(Math.random() * 5) + 1;
  
  if(rand == 1){
    alert("大吉");
  }else if(rand == 2){
    alert("中吉");
  }else if(rand == 3){
    alert("小吉");
  }else if(rand == 4){
    alert("凶");
  }else if(rand == 5){
    alert("大凶");
  }else{
    alert("エラー");
  }
});
